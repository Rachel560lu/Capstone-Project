        function initComparisonSlider() {
            const slider = document.querySelector('.comparison-slider');
            const afterImage = document.querySelector('.comparison-after');
            let isMoving = false;

            slider.addEventListener('mousedown', function(e) {
                e.preventDefault();
                isMoving = true;
            });

            document.addEventListener('mouseup', function() {
                isMoving = false;
            });

            document.addEventListener('mousemove', function(e) {
                if (!isMoving) return;
                
                let x = e.pageX;
                x -= slider.parentNode.getBoundingClientRect().left;
                slideMove(x);
            });

            // è§¦æ‘¸è®¾å¤‡æ”¯æŒ
            slider.addEventListener('touchstart', function(e) {
                e.preventDefault();
                isMoving = true;
            });

            document.addEventListener('touchend', function() {
                isMoving = false;
            });

            document.addEventListener('touchmove', function(e) {
                if (!isMoving) return;
                
                let x;
                if (e.touches) x = e.touches[0].pageX;
                x -= slider.parentNode.getBoundingClientRect().left;
                slideMove(x);
            });

            function slideMove(x) {
                let containerWidth = slider.parentNode.offsetWidth;
                if (x < 0) x = 0;
                if (x > containerWidth) x = containerWidth;
